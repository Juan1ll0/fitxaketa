<script lang="ts">
  import { onMount, onDestroy } from 'svelte';
  import { initAppState, startJornada, stopJornada, getClockedIn, getElapsed, subscribe } from '$lib/stores/app-state';
  import { getAllJornadas, type Jornada } from '$lib/db';
  import { formatearFecha, calcularResumenDia } from '$lib/utils/dashboard';

  export const ssr = false;

  let clockedIn = $state(false);
  let elapsed = $state('00:00:00');
  let hoy = $state('');
  let resumenDia = $state({ totalHoras: 0, totalJornadas: 0 });
  let jornadas = $state<Jornada[]>([]);
  
  let unsubscribe: (() => void) | null = null;
  let fechaInterval: ReturnType<typeof setInterval> | null = null;

  onMount(async () => {
    await initAppState();
    hoy = formatearFecha(new Date());
    
    // Suscribirse a cambios del store
    unsubscribe = subscribe(() => {
      clockedIn = getClockedIn();
      elapsed = getElapsed();
    });
    
    // Verificar cambio de dÃ­a cada minuto
    fechaInterval = setInterval(() => {
      const nuevaFecha = formatearFecha(new Date());
      if (nuevaFecha !== hoy) {
        hoy = nuevaFecha;
        cargarJornadasHoy();
      }
    }, 60000);
    
    await cargarJornadasHoy();
  });

  onDestroy(() => {
    if (unsubscribe) unsubscribe();
    if (fechaInterval) clearInterval(fechaInterval);
  });

  async function cargarJornadasHoy() {
    const todas = await getAllJornadas();
    const inicioHoy = new Date();
    inicioHoy.setHours(0, 0, 0, 0);
    
    jornadas = todas.filter(j => j.start_time >= inicioHoy);
    
    // Calcular resumen incluyendo jornada abierta
    const resumen = calcularResumenDia(jornadas.filter(j => j.status === 'closed'));
    
    // Si hay jornada abierta, sumar tiempo transcurrido
    if (clockedIn) {
      const [h, m, s] = elapsed.split(':').map(Number);
      const minutosTranscurridos = h * 60 + m + s / 60;
      resumen.totalHoras += minutosTranscurridos / 60;
      resumen.totalJornadas += 1; // Contar la jornada abierta
    }
    
    resumenDia = resumen;
  }

  async function handleFichar() {
    if (clockedIn) {
      await stopJornada();
    } else {
      await startJornada();
    }
    await cargarJornadasHoy();
  }
</script>

## Riesgos TÃĐcnicos

### R1: Memory Leak por SuscripciÃģn
**DescripciÃģn:** La suscripciÃģn al store `app-state.ts` puede causar memory leaks si no se limpia al destruir el componente.  
**MitigaciÃģn:** Guardar el `unsubscribe` retornado por `subscribe()` y llamarlo en `onDestroy`.

### R2: CÃĄlculo Incorrecto del Resumen
**DescripciÃģn:** El resumen del dÃ­a debe incluir tanto jornadas cerradas como el tiempo transcurrido de la jornada abierta actual.  
**MitigaciÃģn:** Filtrar jornadas cerradas para `calcularResumenDia`, luego sumar tiempo transcurrido de jornada abierta manualmente en el componente.

**DescripciÃģn:** Si la app permanece abierta al cambiar de dÃ­a, la fecha mostrada puede quedar desactualizada.  
**MitigaciÃģn:** Usar `setInterval` cada 60 segundos para verificar cambio de dÃ­a y actualizar. Limpiar en `onDestroy`.

**DescripciÃģn:** Dexie/IndexedDB no existe en servidor, causando errores de build si SSR estÃĄ habilitado.  
**MitigaciÃģn:** Asegurar `export const ssr = false` en `+page.svelte`.
